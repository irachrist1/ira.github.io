// Define the pins for buttons and the LED
const int button1Pin = 2;
const int button2Pin = 3;
const int ledPin = 13;

// Initialize variables
int delayTime = 1000;  // Initial delay time in milliseconds
int button1State = HIGH;
int button2State = HIGH;
int lastButton1State = HIGH;
int lastButton2State = HIGH;
unsigned long lastDebounceTime = 0;
unsigned long debounceDelay = 50;

void setup() {
  pinMode(button1Pin, INPUT);
  pinMode(button2Pin, INPUT);
  pinMode(ledPin, OUTPUT);
}

void loop() {
  // Read the state of the buttons with debounce
  int reading1 = digitalRead(button1Pin);
  int reading2 = digitalRead(button2Pin);

  if (reading1 != lastButton1State) {
    lastDebounceTime = millis();
  }
  if (reading2 != lastButton2State) {
    lastDebounceTime = millis();
  }

  if (millis() - lastDebounceTime > debounceDelay) {
    if (reading1 != button1State) {
      button1State = reading1;
      if (button1State == LOW) {
        // Increment the delay time
        delayTime += 100;
      }
    }
    if (reading2 != button2State) {
      button2State = reading2;
      if (button2State == LOW) {
        // Decrement the delay time, but ensure it doesn't go below 100 milliseconds
        if (delayTime > 100) {
          delayTime -= 100;
        }
      }
    }
  }

  lastButton1State = reading1;
  lastButton2State = reading2;

  // Blink the LED with the updated delay
  digitalWrite(ledPin, HIGH);
  delay(delayTime);
  digitalWrite(ledPin, LOW);
  delay(delayTime);
}
